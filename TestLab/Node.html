def initialize(*args)
  @ui = TestLab.ui

  @ui.logger.info { "Loading Node '#{self.id}'" }

  super(*args)

  @provider = self.provider.new(self.config, @ui)

  raise NodeError, "You must specify a provider class!" if self.provider.nil?

  @ui.logger.info { "Node '#{self.id}' Loaded" }
end
