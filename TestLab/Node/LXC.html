def lxc(options={})
  if (!defined?(@lxc) || @lxc.nil?)
    @lxc_runner ||= ::LXC::Runner::SSH.new(:ui => @ui, :ssh => self.ssh)
    @lxc        ||= ::LXC.new(:ui => @ui, :runner => @lxc_runner)
  end
  @lxc
end
