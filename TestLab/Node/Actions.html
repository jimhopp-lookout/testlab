def create
  @ui.logger.debug { "Node Create: #{self.id} " }

  (self.state != :not_created) and return false

  please_wait(:ui => @ui, :message => format_object_action(self, 'Create', :green)) do
    @provider.create

    do_provisioner_callbacks(self, :create, @ui)
  end

  true
end
