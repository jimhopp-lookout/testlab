def status
  {
    :id => self.id,
    :instance_id => @provider.instance_id,
    :state => @provider.state,
    :user => @provider.user,
    :identity => @provider.identity,
    :ip => @provider.ip,
    :port => @provider.port,
    :provider => @provider.class,
    :provisioners => self.provisioners.map(&:to_s).collect{ |p| p.split('::').last }.join(',')
  }
end
